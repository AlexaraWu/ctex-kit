\TEST{.../aftername}{
  \begingroup
    \ctexset{
      part/aftername = \par\vskip 50pt,
      chapter/aftername += CHAPTER\sffamily
    }
    \part{部分标题}
    \part*{部分标题}
    \chapter{章标题}
    文字
    \chapter*{章标题}
    文字
    \section{节标题}
    文字
    \section*{节标题}
    文字
    \ctexset{section/aftername+= \quad SECTION}
    \section{节标题}
    文字
    \section*{节标题}
    文字
  \endgroup
  \clearpage
}
